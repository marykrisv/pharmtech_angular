CREATE VIEW currentlocations AS
SELECT l.*, uc.userName as ucUsername, um.userName as umUsername
FROM location l
LEFT JOIN user uc on l.locCreatedBy=uc.userId
LEFT JOIN user um on l.locModifiedBy=um.userId
WHERE l.locDeleted = 0; 

CREATE VIEW currentproducts AS
SELECT p.*, dos.*, str.*, con.*, man.*, res.*, uom.*, uc.userName as ucUsername, um.userName as umUsername
FROM product p
LEFT JOIN dosage dos on p.prodDosId=dos.dosId
LEFT JOIN strength str on p.prodStrId=str.strId
LEFT JOIN concentration con on p.prodConId=con.conId
LEFT JOIN manufacturer man on p.prodManId=man.manId
LEFT JOIN restriction res on p.prodResId=res.resId
LEFT JOIN uom uom on p.prodUomId=uom.uomId
LEFT JOIN user uc on p.prodCreatedBy=uc.userId
LEFT JOIN user um on p.prodModifiedBy=um.userId
WHERE p.prodDeleted = 0;
